function get_products($qty, $location)
{
    global $con;

    // --- STEP 1: DEFINE THE SQL QUERY BASED ON LOCATION ---
    $sql = "SELECT * FROM `ecom_products` "; // Base query

    switch ($location) {
        case "home-trending":
            // Assumes higher ID = Newer/Trending, or use 'product_views' if you have it
            $sql .= "ORDER BY product_id DESC LIMIT $qty";
            break;

        case "home-top-rated":
            // For now, sorting by price as a placeholder for "premium" items
            $sql .= "ORDER BY product_price DESC LIMIT $qty";
            break;

        case "home-new-products-side":
            // Latest additions
            $sql .= "ORDER BY created_at DESC LIMIT $qty";
            break;

        case "shop-page-top":
            // Range example: specific IDs 5 through 10
            $sql .= "WHERE product_id BETWEEN 5 AND 10 LIMIT $qty";
            break;

        case "home-tabs":
            // Randomize items for the tab section
            $sql .= "ORDER BY RAND() LIMIT $qty";
            break;

        case "home-top-selling":
        case "home-top-selling-side":
            // Placeholder: sorting by ID, replace with 'sales_count' if column exists
            $sql .= "ORDER BY product_id ASC LIMIT $qty";
            break;

        case "shop-page":
            // Standard shop view
            $sql .= "ORDER BY product_id DESC LIMIT $qty";
            break;

        default:
            $sql .= "LIMIT $qty";
            break;
    }

    $result = mysqli_query($con, $sql);

    // --- STEP 2: LOOP AND DISPLAY HTML ---
    if (mysqli_num_rows($result) > 0) {

<?php
session_start();

// 1. Initialize the toast message before destroying the existing session variables
$_SESSION['toast-message'] = "Logged out successfully!";
$_SESSION['toast-icon'] = "success";

// 2. Keep the toast variables safe while clearing everything else
$toastMsg = $_SESSION['toast-message'];
$toastIcon = $_SESSION['toast-icon'];

// 3. Unset all of the session variables
$_SESSION = array();

// 4. If it's desired to kill the session entirely, destroy the session cookie as well
if (ini_get("session.use_cookies")) {
    $params = session_get_cookie_params();
    setcookie(
        session_name(),
        '',
        time() - 42000,
        $params["path"],
        $params["domain"],
        $params["secure"],
        $params["httponly"]
    );
}

// 5. Destroy the server-side session file
session_destroy();

// 6. Start a brand new session context just to pass the toast to the index page
session_start();
$_SESSION['toast-message'] = $toastMsg;
$_SESSION['toast-icon'] = $toastIcon;

// 7. Redirect back to the landing page
header("Location: ../index.php"); // Adjust path if index.php is in a different directory
exit();
